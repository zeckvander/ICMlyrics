import React, { useState, useEffect } from "react";
import { Music, Search, RefreshCw } from "lucide-react";
import { Button } from "@/components/ui/button";
import { supabase } from "@/lib/supabaseClient";

export default function SelecionarLouvorModal({ open, onOpenChange, onSelectLouvor }) {
  const [buscaLouvorModal, setBuscaLouvorModal] = useState("");
  const [todosLouvoresBanco, setTodosLouvoresBanco] = useState([]);
  const [carregandoLouvoresBanco, setCarregandoLouvoresBanco] = useState(false);

  useEffect(() => {
    if (open && todosLouvoresBanco.length === 0) {
      carregarLouvores();
    }
  }, [open]);

  const carregarLouvores = async () => {
    setCarregandoLouvoresBanco(true);
    try {
      const { data, error } = await supabase
        .from("louvores")
        .select("id, numero, nome, categoria")
        .order("nome", { ascending: true });

      if (error) throw error;
      setTodosLouvoresBanco(data || []);
    } catch (e) {
      console.error("Erro ao buscar louvores:", e);
    } finally {
      setCarregandoLouvoresBanco(false);
    }
  };

  const louvoresFiltradosBanco = todosLouvoresBanco.filter(l => 
    (l.nome && l.nome.toLowerCase().includes(buscaLouvorModal.toLowerCase())) ||
    (l.numero && String(l.numero).toLowerCase().includes(buscaLouvorModal.toLowerCase())) ||
    (l.categoria && l.categoria.toLowerCase().includes(buscaLouvorModal.toLowerCase()))
  );

  if (!open) return null;

  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/50 p-4">
      <div className="bg-white w-full max-w-md rounded-2xl p-5 shadow-xl space-y-4 max-h-[85vh] flex flex-col">
        <div className="flex items-center justify-between border-b border-slate-100 pb-3">
          <h3 className="text-sm font-bold text-slate-900 flex items-center gap-2">
            <Music className="w-4 h-4 text-indigo-600" /> Selecionar Louvor do Banco
          </h3>
          <button 
            onClick={() => onOpenChange(false)}
            className="text-slate-400 hover:text-slate-600 text-sm font-bold"
          >
            ✕
          </button>
        </div>

        <div className="relative">
          <Search className="w-4 h-4 text-slate-400 absolute left-3 top-2.5" />
          <input
            type="text"
            value={buscaLouvorModal}
            onChange={(e) => setBuscaLouvorModal(e.target.value)}
            placeholder="Pesquisar por nome, número ou categoria..."
            className="w-full pl-9 pr-3 py-2 bg-slate-50 border border-slate-200 rounded-xl text-xs focus:outline-none focus:ring-2 focus:ring-indigo-500"
          />
        </div>

        <div className="flex-1 overflow-y-auto space-y-2 pr-1 max-h-[50vh]">
          {carregandoLouvoresBanco ? (
            <div className="text-center py-8">
              <RefreshCw className="w-6 h-6 animate-spin text-indigo-600 mx-auto" />
              <p className="text-xs text-slate-400 mt-2">Carregando louvores...</p>
            </div>
          ) : louvoresFiltradosBanco.length === 0 ? (
            <p className="text-center text-xs text-slate-400 py-8">Nenhum louvor encontrado.</p>
          ) : (
            louvoresFiltradosBanco.map((louvor) => (
              <div
                key={louvor.id}
                onClick={() => {
                  onSelectLouvor(louvor);
                  onOpenChange(false);
                }}
                className="flex items-center justify-between p-2.5 bg-slate-50 hover:bg-indigo-50/60 border border-slate-100 hover:border-indigo-200 rounded-xl cursor-pointer transition-all"
              >
                <div className="flex items-center gap-2.5 truncate">
                  <span className="font-semibold text-slate-700 bg-white shadow-sm border border-slate-200 px-2 py-0.5 rounded text-[11px] text-center min-w-[36px]">
                    {louvor.numero || louvor.categoria || "—"}
                  </span>
                  <span className="text-xs font-medium text-slate-800 truncate">{louvor.nome}</span>
                </div>
                <span className="text-[10px] text-indigo-600 font-semibold bg-indigo-50 px-2 py-0.5 rounded-full flex-shrink-0">
                  Selecionar
                </span>
              </div>
            ))
          )}
        </div>

        <div className="pt-2 border-t border-slate-100 flex justify-end">
          <Button
            variant="outline"
            onClick={() => onOpenChange(false)}
            className="w-full text-xs h-9 rounded-xl"
          >
            Cancelar
          </Button>
        </div>
      </div>
    </div>
  );
}
